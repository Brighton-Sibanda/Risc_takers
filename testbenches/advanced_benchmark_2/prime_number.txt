srai x3, x1, 1 // divide argument by 2 and loop that many times

loop: // enter for loop
jal x0, mod // call mod function 
modend:
sub x3, x3, x7

mod: // mod function
add x4, x3, x0
add x5, x1, x0
modloop:
sub x5, x5, x4
bge x5, x7, modloop
beq x5, x0, done

prime: // if prime, x2 = 1; else, x2 = 0
